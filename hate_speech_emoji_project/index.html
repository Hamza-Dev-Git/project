""" utils.py Shared helpers for the hate speech + emoji detection pipeline. """ import re OFFENSIVE_WORDS = [ "stupid", "dumb", "idiot", "moron", "ugly", "disgusting", "disgust", "worthless", "pathetic", "useless", "horrible", "awful", "toxic", "vile", "trash", "scum", "loser", "hate", "despise", "freak", "joke", "waste", "sick", "shut up", "get lost", "go away", "vanish", "disappear", "kill", "die", ] HARMFUL_EMOJIS = ["😡", "🤬", "💩", "🖕", "👿", "😤", "🤮", "☠️", "😾", "🔪", "👊", "🤢"] TOKEN_PATTERN = r"[A-Za-z']+|[\U0001F000-\U0001FFFF\u2600-\u27BF]" def clean_text(text: str) -> str: text = str(text).lower().strip() text = re.sub(r"http\S+|www\.\S+", "", text) text = re.sub(r"\s+", " ", text) return text def find_offensive_words(text: str): lower = text.lower() found = [] for word in OFFENSIVE_WORDS: if re.search(r"\b" + re.escape(word) + r"\b", lower): found.append(word) return sorted(set(found)) def find_harmful_emojis(text: str): found = [e for e in HARMFUL_EMOJIS if e in text] return sorted(set(found)) def highlight_html(text: str, offensive_words, harmful_emojis) -> str: result = text for emoji in harmful_emojis: result = result.replace(emoji, f'{emoji}') for word in sorted(offensive_words, key=len, reverse=True): pattern = re.compile(r"\b(" + re.escape(word) + r")\b", re.IGNORECASE) result = pattern.sub(r'\1', result) return result
